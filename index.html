less 
// 关键需要 less-plugin-functions 来给 less 提供自定义函数支持
// You need less-plugin-functions
.function {
    .rem(@size) {
        // 640 是设计稿宽度
        // 640 is psd width
        return: @size/(640/320*16)+0rem;
    }
}
css 
@import "./rem";

    box-shadow: rem(11) rem(22) rem(33) pink;
